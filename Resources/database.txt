CREATE TABLE Patient (
	pID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	date_of_birth TEXT NOT NULL,
	f_name TEXT NOT NULL,
	l_name TEXT NOT NULL,
	email TEXT NOT NULL
	);
	
CREATE TABLE Place_of_service (
	placeID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	place_description TEXT
	);
	
CREATE TABLE Room (
	roomID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	room_description TEXT
	);
	
CREATE TABLE Doctor (
	dID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	f_name TEXT NOT NULL,
	l_name TEXT NOT NULL,
	email TEXT,
	type INTEGER NOT NULL
	);
	
CREATE TABLE Appointment (
	aptID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	pID INTEGER,
	date TEXT,
	placeID TEXT,
	roomID TEXT,
	code_type INTEGER,
	complete INTEGER,
	FOREIGN KEY (pID) REFERENCES Patient (pID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (placeID) REFERENCES Place_of_service (placeID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (roomID) REFERENCES Room (roomID) ON DELETE CASCADE ON UPDATE CASCADE
	);
	
CREATE TABLE Has_doc (
	aptID INTEGER NOT NULL,
	dID INTEGER NOT NULL, 
	PRIMARY KEY (aptID,dID),
	FOREIGN KEY (aptID) REFERENCES Appointment (aptID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (dID) REFERENCES Doctor (dID) ON DELETE CASCADE ON UPDATE CASCADE
	);
	
CREATE TABLE Apt_type (
	apt_code TEXT PRIMARY KEY NOT NULL,
	type_description TEXT,
	code_description TEXT
	);
	
CREATE TABLE Modifier (
	modifierID INTEGER PRIMARY KEY NOT NULL,
	modifier TEXT NOT NULL,
	modifier_description TEXT NOT NULL
	);
	
CREATE TABLE Has_modifiers (
	aptID INTEGER NOT NULL,
	apt_code TEXT NOT NULL,
	modifierID INTEGER NOT NULL,
	PRIMARY KEY (aptID, apt_code, modifierID),
	FOREIGN KEY (apt_code) REFERENCES Apt_type (apt_code) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (modifierID) REFERENCES Modifier (modifierID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (aptID) REFERENCES Appointment (aptID) ON DELETE CASCADE ON UPDATE CASCADE
	);

CREATE TABLE ICD9_condition (
	ICD9_code TEXT PRIMARY KEY NOT NULL,
	condition_name TEXT NOT NULL
	);

CREATE TABLE ICD10_condition (
	ICD10_ID INTEGER PRIMARU KEY NOT NULL,
	ICD10_code TEXT NOT NULL,
	description_text TEXT
	);
	
CREATE TABLE Characterized_by (
	ICD10_ID INTEGER,
	ICD9_code TEXT,
	PRIMARY KEY (ICD10_ID, ICD9_code),
	FOREIGN KEY (ICD10_ID) REFERENCES ICD10_condition (ICD10_ID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (ICD9_code) REFERENCES ICD9_condition (ICD9_code) ON DELETE CASCADE ON UPDATE CASCADE
	);
	
CREATE TABLE Extension (
	Extension_code TEXT NOT NULL,
	ICD10_ID INTEGER NOT NULL,
	Extension_description TEXT NOT NULL,
	PRIMARY KEY (Extension_code, ICD10_ID),
	FOREIGN KEY (ICD10_ID) REFERENCES ICD10_condition (ICD10_ID) ON DELETE CASCADE ON UPDATE CASCADE
	);
	
CREATE TABLE Has_type (
	aptID INTEGER NOT NULL,
	apt_code TEXT NOT NULL,
	ICD10_ID TEXT NOT NULL,
	visit_priority INTEGER NOT NULL,
	icd_priority INTEGER NOT NULL,
	extension TEXT,
	PRIMARY KEY (aptID, apt_code, ICD10_ID),
	FOREIGN KEY (aptID) REFERENCES Appointment (aptID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (apt_code) REFERENCES Apt_type (apt_code) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (ICD10_ID) REFERENCES ICD10_condition (ICD10_ID) ON DELETE CASCADE ON UPDATE CASCADE
	);
	
CREATE TABLE Condition_location (
	LID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	location_name TEXT
	);
	
CREATE TABLE Sub_location (
	LID INTEGER NOT NULL,
	parent_locationID INTEGER NOT NULL,
	PRIMARY KEY (LID, parent_locationID),
	FOREIGN KEY (LID) REFERENCES Condition_location (LID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (parent_locationID) REFERENCES Condition_location(LID) ON DELETE CASCADE ON UPDATE CASCADE
	);
CREATE TABLE Located_in (
	ICD10_ID INTEGER NOT NULL,
	LID INTEGER NOT NULL,
	PRIMARY KEY (ICD10_ID, LID),
	FOREIGN KEY (ICD10_ID) REFERENCES ICD10_condition (ICD10_ID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (LID) REFERENCES Condition_location (LID) ON DELETE CASCADE ON UPDATE CASCADE
	);
	
CREATE TABLE Module (
	MID INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
	module_name TEXT NOT NULL
	);
	
CREATE TABLE Contains_location (
	MID INTEGER NOT NULL,
	LID INTEGER NOT NULL,
	PRIMARY KEY (MID,LID),
	FOREIGN KEY (LID) REFERENCES Condition_location (LID) ON DELETE CASCADE ON UPDATE CASCADE,
	FOREIGN KEY (MID) REFERENCES Module (MID) ON DELETE CASCADE ON UPDATE CASCADE
	);
